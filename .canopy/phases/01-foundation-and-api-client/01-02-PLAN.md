  <name>Error Classification and Retry-After Parsing</name>
  <files>src/api/errors.ts, src/api/__tests__/errors.test.ts</files>
  <behavior>
    classifyError(error) behavior:
    - OpenAI.APIError with status 429 -> { type: 'rate_limited', statusCode: 429, retryAfterMs: parsed, message }
    - OpenAI.APIError with status 500 -> { type: 'server_error', statusCode: 500, message }
    - OpenAI.APIError with status 503 -> { type: 'server_error', statusCode: 503, message }
    - OpenAI.APIError with status 400 -> { type: 'client_error', statusCode: 400, message }
    - OpenAI.APIError with status 401 -> { type: 'client_error', statusCode: 401, message }
    - OpenAI.APIConnectionError -> { type: 'timeout', message }
    - Generic Error with 'timeout' in message -> { type: 'timeout', message }
    - Generic Error with 'ETIMEDOUT' in message -> { type: 'timeout', message }
    - Unknown error -> { type: 'client_error', message: String(error) }

    parseRetryAfter(headerValue) behavior:
    - "5" -> 5000 (seconds to milliseconds)
    - "120" -> 120000
    - "Thu, 01 Jan 2099 00:00:00 GMT" -> future date in ms from now (positive number)
    - null/undefined -> undefined
    - "invalid" -> 1000 (fallback 1 second)
    - "" -> 1000 (fallback)
  </behavior>
